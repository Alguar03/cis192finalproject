<script>
    function startSearch()
    {
        var street      = $("#street").val();
        var city        = $("#city").val();
        var region      = $("#region").val();
        var postal_code = $("#postal_code").val();
        var url         = !isBatch ? $("#url").val() : '';

        // do some validation... BEFORE we submit the form and create accounts and things...
// NOTE: we don't use the jquery validator here since I don't know how to make the url optional, and I can't seem to get it to use OUR fields for the error messages. 
        failed = false;
        // Note since there is more than one general error on this form, we need to specify the form as well.
        $("#startSearch #general_error").html('');
        $("#error_street").html('');
        $("#error_city").html('');
        $("#error_region").html('');
        $("#error_postal_code").html('');
        $("#error_url").html('');

        if (street == '')
        {
            $("#error_street").html("Please enter your street.");
            failed = true;
        }

        if (city == '')
        {
            $("#error_city").html("Please enter your city.");
            failed = true;
        }
        if (region == 'defaultUSA')
        {
            $("#error_region").html("Please choose a region.");
            failed = true;
        }

        postalCodeFailed = false;
        if (postal_code == '')
        {
            $("#error_postal_code").html("Please enter your postal code.");
            failed = true;
            postalCodeFailed = true;
        }
        if (!postalCodeFailed && !isValidPostalCode(postal_code))
        {
            $("#error_postal_code").html("Please enter a valid postal code (ex: 02138).");
            failed = true;
        }

        if (url != '')
        {
            if (!isValidURL(url))
            {
                $("#error_url").html("Please enter a valid URL.");
                failed = true;
            }
        }

        if (failed)
        {
            $("#startSearch #general_error").html('Please correct errors in red before continuing.');
            return false;
        }

        var campaignTitle = getBizName();
        var bizLocation = getBizLocation();

        var phoneNumber = ""; // Has always been empty
        var lightbox = $("#lightbox").val();

        if (isGroup || isBatch)
        {
            validateCampaignLocation(street, city, region, postal_code, url);
        }
        else
        {
            if (user_id > 0)
            {
                // Since the user enters all the data, replicate the campaignTitle as the businessName.
                addCampaign(campaignTitle, street, city, region, postal_code, campaignTitle, phoneNumber, bizLocation, lightbox, url);
            }
            else
            {
                var email = $("#reg_email").val();
                autoCampaign(email, campaignTitle, street, city, region, postal_code, campaignTitle, phoneNumber, bizLocation, lightbox, url);
            }
        }
    }
</script>
