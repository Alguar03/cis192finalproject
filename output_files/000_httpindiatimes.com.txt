function select_topic(topic_id,action) {
     if(action=='keyup') {
        $("#li_"+topic_id).removeClass('active');
        if(typeof($("#li_"+(topic_id-1))) != 'undefined') {
		    $("#li_"+(topic_id-1)).addClass('active');
        } else {
            selected_topic=the_global_topic;
        }
    } else if(action=='keydown') {
		$("#li_"+topic_id).addClass('active');
        if(typeof($("#li_"+(topic_id-1))) != 'undefined'){
			$("#li_"+(topic_id-1)).removeClass('active');
        } else {
			selected_topic=1;
        }        
    }
}

			$('#' + to_be_active).attr('class', 'a'+ to_be_active + ' active');
			if(to_be_active==12){
				$('#rightarr').removeClass('enable');
			}
			if(to_be_active > 1){
				$('#leftarr').attr('class', '');
				$('#leftarr').attr('class', 'lftarrow enable');
			}
		}
	}else if(flag=='prev'){
		if(current_active == 1){
			send_request = 0;
		}else{
			to_be_active = eval(current_active-1);  //alert(to_be_active);
			if(to_be_active==1){
				$('#leftarr').removeClass('enable');
				send_request = 0;	
			}else if(to_be_active!=12){
				$('#rightarr').attr('class', '');
				$('#rightarr').attr('class', 'rghtarrow enable');
			}
			$('#' + current_active).removeClass('active');
			$('#' + to_be_active).attr('class', 'a'+ to_be_active + ' active');
		}
	}else if(parseInt(flag)){
		to_be_active = parseInt(flag); 
		$('#' + current_active).removeClass('active');
		$('#' + to_be_active).attr('class', 'a'+ to_be_active + ' active');
		$('#leftarr').attr('class', 'lftarrow enable');
		$('#rightarr').attr('class', 'rghtarrow enable');
		if(to_be_active == 1){
			$('#leftarr').removeClass('enable');
		}else if(to_be_active == 12){
			$('#rightarr').removeClass('enable');	
		}
	}
	if(send_request){
		getAstroData(to_be_active);
	}
}
function getAstroData(to_be_active){
	var selected_sign = 1;
	var data_string = 'contenttype=1&timetype=1';
	var url = '/astrofeeds.php'; //alert(to_be_active);
	if(typeof(to_be_active) != 'undefined'){
		selected_sign = to_be_active;
	}
	data_string = data_string + '&sunsign=' +selected_sign+ '&astroid=1';
	$.ajax({
		type: "POST",
		url: url,
		data: data_string,
		success: function(resultdata){
			$('.AstroDetail').html(resultdata);
		}
	});
}
