<script type="text/javascript">
// Launch video player with the following configuration:
// - xml: Path to XML configuration file
// - chapter: ID of deep linked chapter (defined in config file) to play 
function embedPlayer(xml, chapter) {
	// Example: embedPlayer('/us_non_production_assets/flash/player/xml/player_config.xml','picturequality')
	var flashvars = {};
	flashvars.config_file = xml;
    flashvars.main_file = "http://images.delta.com.edgesuite.net/movies/videoplayer.swf";
	flashvars.css_file = "http://images.delta.com.edgesuite.net/movies/css/flash.css";
	flashvars.fonts_xml = "/shared/components/movies/xml/player_styles.jsp";
    flashvars.debug = "false";
    // possible values are: verbose, warning, error
    flashvars.log_level = "verbose";
	
	// Make player
	makePlayer(flashvars);
}

function getFlashVersion(){
  // Flash detection on IE
  try {
    try {
      var axo = new ActiveXObject('ShockwaveFlash.ShockwaveFlash.6');
      try { axo.AllowScriptAccess = 'always'; }
      catch(e) { return '6,0,0'; }
    } catch(e) {}
    return new ActiveXObject('ShockwaveFlash.ShockwaveFlash').GetVariable('$version').replace(/\D+/g, ',').match(/^,?(.+),?$/)[1];
  // Flash detection on other browsers
  } catch(e) {
    try {
      if(navigator.mimeTypes["application/x-shockwave-flash"].enabledPlugin){
        return (navigator.plugins["Shockwave Flash 2.0"] || navigator.plugins["Shockwave Flash"]).description.replace(/\D+/g, ",").match(/^,?(.+),?$/)[1];
      }
    } catch(e) {}
  }
  return false;
}


function makePlayer(flashvars) {
	var params = {};
	params.allowScriptAccess = "always";
	params.wmode = "opaque";
	params.allowFullScreen = "true";
	
	var attributes = {};
	attributes.id = "player";
	attributes.name = "player";
	
	document.getElementById("player_container").innerHTML = "<div id='player'></div>";
	if (getFlashVersion()!=false) {
		// Embed player
		swfobject.embedSWF("http://images.delta.com.edgesuite.net/movies/videoplayer_preloader.swf", "player", "896", "504", "9.0.115", "http://images.delta.com.edgesuite.net/movies/expressInstall.swf", flashvars, params, attributes);
	} else {
		// Show backup image
		document.getElementById("player").innerHTML = "<img border='0' src='http://images.delta.com.edgesuite.net/movies/flash_error_message_60k.jpg' usemap='#noflashmap' />";
	}
	// Unhide the Flash and grey out the rest of the page
	document.getElementById("player_embedlayer").className = "";
		if(get("preferenceItinId")){
			hide("preferenceItinId");
		}
	
	// Calculate height of page to set size over overlay (to deal with scrolling)
	windowHeight = window.innerHeight ? window.innerHeight : document.documentElement.clientHeight ? document.documentElement.clientHeight : document.body.clientHeight;
	pgHeight = Math.max(Math.max(document.body.scrollHeight, windowHeight), windowHeight);
	
	// Hide the dropdowns to solve issue of them showing through the flash.  These will be set to visible in the closePlayer function
	var allSelects = document.getElementsByTagName('select');
	for(var n=0;n<allSelects.length;n++){ allSelects[n].style.visibility="hidden"; }
	
	document.getElementById("player_overlay").style.height = pgHeight+"px";
	document.getElementById("player_overlay").className = "";
}

// THIS METHOD MUST BE IMPLEMENTED
// This function is called by the Flash to close the layer the player is in
function closePlayer() {
	// Display the page again and close the Flash layer
	document.getElementById("player_overlay").className = "player_hide";
	document.getElementById("player_embedlayer").className = "player_hide";
	// Destroy the Flash
	document.getElementById("player_container").innerHTML = "";
			if(get("preferenceItinId")){
			get("preferenceItinId").style.display = '';
		}
	// Show the dropdowns that were hidden in the makePlayer function
	var allSelects = document.getElementsByTagName('select');
	for(var n=0;n<allSelects.length;n++){ allSelects[n].style.visibility="visible"; }
}
