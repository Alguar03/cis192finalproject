			return; 
		}
		trib_debug('Queue Length LOW do lookup');
		new Ajax.Request('/ticker/fetch_next/' + last_date, {
	        method: "get",
	        onComplete: function(request){
	            if (request.status == 200) { 
					o = request.responseText.evalJSON();
					if (o != null) {
						trib_debug('Entry added to Queue');
                        queue.push(o);
						last_date = o.time;
					}	
				} 
	       	}
	    });
	}
	
	var beat_index = 1;
    var pause = false;
	
	function move_scroll(){
		if (pause || first_entity == null || queue.length == 0) {return;}
		
		//trib_debug("moveScroll - " + margin_value);
		
		margin_value = margin_value + 1;
		first_entity.morph('margin-left:' + margin_value + 'px;', {duration: 0});
		
		//trib_debug(first_entity.getStyle('margin-left'));
		
		if (margin_value == 0 ) {
			update_ticker();
		}
	}
	
	function update_ticker() {
		if (queue.length == 0 || pause == true) {  
			setTimeout('update_ticker();', ticker_delay)
			return;	
		}
		trib_debug('Queue Entries ' + queue.length);
		entry = queue.shift();
		if (entry) {
			str = '<div class="beat_window" style="width:350px;" id="beat_wi_' + beat_index + '"><div id="beat_no_' + beat_index + '" class="beat">' + entry.text + '</div></div>';
			$('ticker').insert({
				top: str
			});
			beat_in(beat_index);
			beat_index++;
			show_dates_as_local_time();
		}		
	}
	
    function beat_in(beat_no) {
		margin_value = -350;
		first_entity = $('beat_wi_' + beat_no);
	}    
	Event.observe(window, 'load', ticker_date_init);	
		
    var ticker_active_updater;
	function ticker_date_init() {

		show_dates_as_local_time();
		ticker_active_updater = setInterval("show_dates_as_local_time();", 10000);
		ticker_active_updater = setInterval("update_queue();", 500);
		update_ticker();
		setInterval("move_scroll();", ticker_delay);
	}
</script>
