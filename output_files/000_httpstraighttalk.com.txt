	function changeCountrySelection () {
		if($id("country_list")) {
			var country_value = $id("country_list").value;
			var state_list = $id("state_list");
			var state_box = $id("state_box");
			var prov_box = $id("prov_list");
			if (country_value == "USA") {
				state_list.disabled = false;
				removeClass(state_list,"hide");
				
				state_box.disabled = true;
				addClass(state_box,"hide");
				
				prov_box.disabled = true;
				addClass(prov_box,"hide");
				
				addClass($id("country_warning_msg"),"hide");
				removeClass($id("phone"), "maskIntlMin");
				addClass($id("phone"), "maskMin");
				DMasked.attach();
			} else if (country_value == "Canada") {
				prov_box.disabled = false;
				removeClass(prov_box,"hide");
				
				state_list.disabled = true;
				addClass(state_list,"hide");
				
				state_box.disabled = true;

				removeClass($id("country_warning_msg"),"hide");
				removeClass($id("phone"), "maskMin");
