    function getNameAndYear(firstName, lastName, year){
        var firstNameMaxLength = 12;
        var lastNameMaxLength = 12
        var nameString = "";
        var overflow = 0;
        if(year != ''){ lastNameMaxLength = 9; }
        if(firstName.length > firstNameMaxLength){
            // first name too long
            nameString += firstName.substring(0, firstNameMaxLength-1) + "-<br/>";
            if(firstName.length > (firstNameMaxLength + lastNameMaxLength)){
                // first name longer than both lines combined
                nameString += firstName.substring(firstNameMaxLength-1, firstNameMaxLength + lastNameMaxLength - 3);
            }
            else{
                nameString += firstName.substring(firstNameMaxLength-1) + " ";
            }
            overflow = firstName.length - firstNameMaxLength + 1 + 1; // add 1 for the '-' and 1 for the space that will have to be inserted on the next line
        }
        else{
            nameString += firstName + "<br/>";
        }
        if(overflow > lastNameMaxLength){
            // first name took full 2 lines so just add ... and class
            nameString += '...';
        }
        if(lastName.length + overflow > lastNameMaxLength){
            nameString += lastName.substring(0, lastNameMaxLength-overflow-2) + "...";
        }
        else{
            nameString += lastName;
        }
        if(year != ''){
            nameString += " '" + year;
        }
        return nameString;
    }
