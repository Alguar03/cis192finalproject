	  function twitterEventHandler(event){
		if (event) {
		  var targetUrl; // Default value is undefined.
		  if (event.target && event.target.nodeName == 'IFRAME') {
			targetUrl = _ga.extractParamFromUri_(event.target.src, 'url');
		  }
		  _gaq.push([trackerName + '_trackSocial', 'twitter', event.type,
			targetUrl, opt_pageUrl]);
		}
	  }
	
	  var trackerName = _ga.buildTrackerName_(opt_trackerName);
	  try {
		if (twttr && twttr.events && twttr.events.bind) {
		  twttr.events.bind('tweet', twitterEventHandler);
		  twttr.events.bind('follow', twitterEventHandler);
		  twttr.events.bind('unfollow', twitterEventHandler);
		}
	  } catch (e) {}
	};


	_ga.extractParamFromUri_ = function(uri, paramName) {
	  if (!uri) {
		return;
	  }
	  var uri = uri.split('#')[0];  // Remove anchor.
	  var parts = uri.split('?');  // Check for query params.
	  if (parts.length == 1) {
		return;
	  }
	  var query = decodeURI(parts[1]);

	  // Find url param.
	  paramName += '=';
	  var params = query.split('&');
	  for (var i = 0, param; param = params[i]; ++i) {
		if (param.indexOf(paramName) === 0) {
		  return unescape(param.split('=')[1]);
		}
	  }
	  return;
	};
