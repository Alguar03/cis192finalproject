	if (valid_email.test($('#signup_email_address').attr('value')) == false) {

		$('#signup_email_address').focus();

	} else {
		$('#signup_email_form_address').hide();
		$('#signup_email_form_password').fadeIn();
	}
	e.preventDefault();
});

	var form = $(this);
	if ($('#signup_password').attr('value') != $('#signup_password_verify').attr('value')) {
		$('#signup_email_form_password .alert').fadeIn();
		$('#signup_password').focus();

	} else {
		$.post(form.attr('action'), form.serialize(), function(data) {
			$('#signup_email_form_password').hide();
			if ($(data).find('.messages.error').length) {
				$('#signup_email_form_address').fadeIn();
				$('#signup_email_form_address .alert').html($(data).find('.messages.error').html()).fadeIn();

			} else {
				$('#signup_email_form_success').fadeIn();
			}
		});
	}
	e.preventDefault();
});
