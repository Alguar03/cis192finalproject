            var input = document.getElementById('search_input');
            var autocomplete = new google.maps.places.Autocomplete(input);
            autocomplete.setTypes([]);
            
            google.maps.event.addListener(autocomplete, 'place_changed', function() {
                $('#search_form').submit();
            });
            
            $('#search_form').submit(function() {
                var address = $('#search_input').val().toLowerCase().replace(/\s+/g, ' ').replace(/^\s+|\s+$/g, '');
                var geocoder = new google.maps.Geocoder();
                var start_time =(new Date()).getTime();
                
                if (address.length != 0 && address != 'Enter your delivery address.'.toLowerCase()) {
                    geocoder.geocode({ address: address, region: 'US' }, function(results, status) {
                        if (status == google.maps.GeocoderStatus.OK && status != google.maps.GeocoderStatus.ZERO_RESULTS) {
                            var formatted_address = results[0].formatted_address;
                            var arrAddress = formatted_address.split(', ');
                            var count = arrAddress.length;

                            if (arrAddress[count - 1] == 'USA') {
                                if (count > 2){ //looks like an address

                                    var state_zip = arrAddress[count - 2];
                                    if ((state_zip.substr(state_zip.length - 5, 5)).match(/^\d+$/)){ //isnumeric
                                        var state = state_zip.substr(0, 2);
                                        var zip_code = state_zip.substr(state_zip.length - 5, 5);
                                    }else{
                                        var state = state_zip;
                                        var zip_code = '';
                                    }

                                    var city = arrAddress[count - 3];
                                    if (count > 3) {
                                        var street_address = arrAddress[0];
                                    }else{
                                        var street_address = '';
                                    }
                                }
                                var latitude = results[0].geometry.location.lat();
                                var longitude = results[0].geometry.location.lng();
                                var location_type = results[0].geometry.location_type;

                                var json_data = {
                                    "version":"1",
                                    "type":"info",
                                    "address_full":formatted_address.replace(", USA", ""),
                                    "street_address":street_address,
                                    "city":city,
                                    "state":state,
                                    "zip_code":zip_code,
                                    "location_type":location_type,
                                    "latitude":latitude.toString(),
                                    "longitude":longitude.toString(),
                                    "address_id":"0",
                                    "customer_delivery_id":null
                                };

                                $.post("/localapi/address/", { query: $.toJSON(json_data) }, function(data) {
                                        var urldata = eval('(' + data + ')');
                                        var urlname = urldata.address[0].urlname;

                                        // GA event for form submission that appears to be successful
                                        var end_time =(new Date()).getTime();
                                        _gaq.push(['_trackEvent', 'home', 'timer', 'address_lookup', (end_time - start_time), true]);
                                        _gaq.push(['_trackEvent', 'home', 'submit', formatted_address, 0, true]);

                                        location.href = '/custom-results/online-ordering/' + urlname + '/';
                                    }	
                                );
                            } else {
                                _gaq.push(['_trackEvent', 'home', 'submit', 'failed_non_usa', 0, true]);
                                _gaq.push(['_trackEvent', 'home', 'country_desired', arrAddress[count - 1], 0, true]);
                                alert("We're sorry.  We only are working with locations in the U.S.A. at this time.  Your request has been noted and we will expand to other countries at some time in the future.");
                                return false;
                            }
                        } else {
                            _gaq.push(['_trackEvent', 'home', 'submit', 'failed_unknown_address', 0, true]);
                            alert("We and our friends at Google Maps could not find a location matching that entry. Please try again.");
                            return false;
                        }
                    });
                } else {
                    _gaq.push(['_trackEvent', 'home', 'submit', 'submit_wo_value', 0, true]);
                    alert("Please enter your delivery address.");
                }
                
                return false;
            });
