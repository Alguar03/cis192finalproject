  var timeout;
  var total;

  function selectarticle(num){
    for(a=0; a<=total; a++){
      if(a==num){
       document.getElementById("featured-"+a).style.display="block";
       document.getElementById("feature_bt_"+a).style.color="#FFD457";

        document.getElementById("featured-"+a).style.display="none";
        document.getElementById("feature_bt_"+a).style.color="#FFFFFF";

    }
  }

  function slideshow(num){
    selectarticle(num);
    prevcolor="#FFD457"
    if(num<total)
      num+=1;
    else
      num=0;
    clearTimeout(timeout);
    timeout=setTimeout('slideshow('+num+')',5000);
  }

  function stop(){
    clearTimeout(timeout);
  }
