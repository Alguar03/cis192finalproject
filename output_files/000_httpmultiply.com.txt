var AllCalendars = new Array();
var Today = new Date();
var WeekDays = new Array('S','M','T','W','R','F','S');
var MonthDays = new Array(31,28,31,30,31,30,31,31,30,31,30,31);
var MonthNames = new Array('January','February','March','April','May','June','July','August','September','October','November','December');

function toggle_calendar_widget(objName, onclick_function) {
    if (!onclick_function) {
	onclick_function = '';
    }
    var obj = getEl(div_pre + objName);

    if (!obj.calendar) {
	obj.calendar = new calendar(onclick_function);
	obj.calendar.id = objName;
	AllCalendars.push(objName);
	set_calendar(obj.calendar.id);
	obj.calendar.populate_dates();
    }
    
    var vis = obj.calendar.visibility;

    // Clear all other calendars
    for (x=0; x<AllCalendars.length; x++) {
	var tmpcal = getEl(div_pre + AllCalendars[x]).calendar;
	tmpcal.setVisibility('none');
    }
    
    if (vis == 'block') {
	obj.calendar.setVisibility('none');
    } else { 
	obj.calendar.setVisibility('block');
    }
}

function calendar(onclick_function) {
    this.date = Today;
    this.visibility = 'none';
    this.month = function () { return this.date.getMonth(); };
    this.day_of_week = function () { return this.date.getDay(); };
    this.day_of_month = function () { return this.date.getDate(); };
    this.year = function () { return (this.date.getYear() < 1000) ? this.date.getYear() + 1900 : this.date.getYear(); };
    this.hour = function () { return this.date.getHours(); };
    this.seconds = function () { return this.date.getSeconds(); };
    this.minutes = function () { return this.date.getMinutes(); };
    this.setVisibility = function(vis) {
	getEl(div_pre + this.id).style.display = vis;
	this.visibility = vis;
    };

    this.populate_dates = function () {
	getEl(month_pre + this.id).innerHTML = (getEl(this.id + '_lang').value == 'ja') ? this.year() + " " + MonthNames[this.month()] : MonthNames[this.month()] + " " + this.year();
	var startDay = this.day_of_week() - ((this.day_of_month()-1) % 7);
	startDay = (startDay < 0) ? startDay + 7 : startDay;
	var add = 0;
	currentCell = startDay;
	if (this.month() == 1 && !(this.year() % 4)) { add = 1; }
	for (x=1; x<=(MonthDays[this.month()] + add); x++) {
	    var contents = '<a href="javascript:set_calendar_day(\'' + x + '\', \'' + this.id + '\'); ' + onclick_function + '">' + x + '</a>';
	    getEl(day_pre + currentCell + "_" + this.id).innerHTML = contents;
	    currentCell++;
	}
	if (startDay > 0) {
	    for (x=0; x<startDay; x++) {
		getEl(day_pre + x + "_" + this.id).innerHTML = '&nbsp;';
	    }    
	}
	if (currentCell < 41) {
	    for (x=currentCell; x<=41; x++) {
		getEl(day_pre + x + "_" + this.id).innerHTML = '&nbsp;';
	    }
	}
	if (currentCell <= 35) { 
	    getEl(row_pre + "6_" + this.id).style.display = 'none';
	}
	else {
	    getEl(row_pre + "6_" + this.id).style.display = '';
	}
	if (currentCell <= 28) {
	    getEl(row_pre + "5_" + this.id).style.display = 'none';
	}
	else {
	    getEl(row_pre + "5_" + this.id).style.display = '';
    	}
    };
    this.sub_mon = function () {
	if (this.month() <= 0) { 
	    if ((this.year() - 1) >= getEl(this.id + '_yearmin').value) {
		this.date.setYear(this.year() - 1); 
		this.date.setMonth(11); 
	    }
	}
	else { this.date.setMonth(this.month() - 1); }
	update_dropdowns(this.id,this); 
    };
    this.add_mon = function () {
	if (this.month() >= 11) {
	    if ((this.year() + 1) <= getEl(this.id + '_yearmax').value) {
		this.date.setYear(this.year() + 1); 
		this.date.setMonth(0); 
	    }
	}
	else { this.date.setMonth(this.month() + 1); }
	update_dropdowns(this.id,this); 
    };
    this.set_day_of_month = function (day) {
	this.date.setDate(day);
    };
    this.set_year = function (year) {
	this.date.setYear(year);
    };
    this.set_month = function (month) {
	this.date.setMonth(month);
    };
}

function subtract_month(cal) {
    var obj = getEl(div_pre + cal);
    obj.calendar.sub_mon();
    obj.calendar.populate_dates();
}

function add_month(cal) {
    var obj = getEl(div_pre + cal);
    obj.calendar.add_mon();
    obj.calendar.populate_dates();
}

function set_calendar_day(day,cal) {
    var obj = getEl(div_pre + cal);
    obj.calendar.set_day_of_month(day);
    obj.calendar.setVisibility('none');
    update_dropdowns(obj.calendar.id,obj.calendar); 
}

function update_dropdowns(id,cal) {
    dayobject = getEl(id + "_day");
    monthobject = getEl(id + "_mon");
    yearobject = getEl(id + "_year");
    for (x=0; x<dayobject.options.length; x++) {
	if (dayobject.options[x].value == cal.day_of_month()) {
	    dayobject.selectedIndex = x;
	}
    } 
    for (x=0; x<monthobject.options.length; x++) {
	if (monthobject.options[x].value == (cal.month()+1)) {
	    monthobject.selectedIndex = x;
	}
    } 
    for (x=0; x<yearobject.options.length; x++) {
	if (yearobject.options[x].value == cal.year()) {
	    yearobject.selectedIndex = x;
	}
    } 
    if (id == 'start_date') {
	set_start_date();
    }
}

function set_calendar(name) {
    var obj = getEl(div_pre + name);
    // get info from drop downs, as they have changed
    dayobject = getEl(name + "_day");
    monthobject = getEl(name + "_mon");
    yearobject = getEl(name + "_year");
    if (obj.calendar) {
	obj.calendar.set_day_of_month(dayobject.options[dayobject.selectedIndex].value);
	obj.calendar.set_month(monthobject.options[monthobject.selectedIndex].value - 1);
	obj.calendar.set_year(yearobject.options[yearobject.selectedIndex].value);
	obj.calendar.populate_dates();
    }
}

function create_div(name) {
    if (!getEl(div_pre + name)) {
	var div = document.createElement('div');
	div.style.display = 'none';
	div.id = div_pre + name;
	div.innerHTML = display_calendar_template(name);
	getEl(name + '_calwidget').appendChild(div);
    }
}

function multiply_tracker() {
    this.segmentation_variables = '';
    this.main_tracker;
    this.group_info = new Array;
    this.group_trackers = new Array;
    this.deferred_tracks = new Array;
    this.seller_tracker;
    this.region_tracker;
    this.region_tracking_code = "UA-23414905-1";
    this.enable_seller_tracker = function () { 
       this.seller_tracker_enabled = true;
    }
    this.enable_region_tracker = function () { 
	if( this.region_tracking_code != '' ){
	    this.region_tracker_enabled = true;
	}
    }
    this.add_group_tracker = function (name,profile) {
	this.group_info.push({name:    name,
	                      profile: profile});
    };
    this.init_trackers = function () {
	// Set up main tracker
	this.main_tracker = _gat._createTracker("UA-82472-1", "main");
	this.main_tracker._setDomainName("multiply.com");
	this.main_tracker._addIgnoredRef("multiply.com");
	
        if (this.seller_tracker_enabled) {
            this.seller_tracker = _gat._createTracker("UA-18197475-1", 'commerce');
            this.seller_tracker._setDomainName("multiply.com");
            this.seller_tracker._setAllowHash(true);
            this.seller_tracker._addIgnoredRef("multiply.com");
        }
	if (this.region_tracker_enabled) {
            this.region_tracker = _gat._createTracker(this.region_tracking_code, 'buyer');
	    this.region_tracker._setDomainName("multiply.com");
            this.region_tracker._addIgnoredRef("multiply.com");
	}
	// Set up group trackers
	for (i in this.group_info) {
	    this.group_trackers[i] = _gat._createTracker(this.group_info[i].profile, 'groups');
	    this.group_tracker[i]._addIgnoredRef("multiply.com");
	}
	for (i in this.deferred_tracks) {
	    var track = this.deferred_tracks[i];
	    this.track_url(track.url,track.detail_url,track.extension);
	}
    };
    this.track_url = function (url,detail_url,extension) {
	//console.log('tracking '+url+' - '+detail_url+' - '+extension);
	if (!url) return;
	if (this.main_tracker) {
	    var track_url = url;
	    if (extension) {
		track_url = track_url + ' (' + extension + ')';
	    }
	    track_url = track_url + '?' + this.segmentation_variables;
	    this.main_tracker._trackPageview(track_url);
	}
	else {
	    this.deferred_tracks.push({url: url, detail_url: detail_url, extension: extension});
	}
	if (this.region_tracker) {
	    var track_url = url;
	    if (extension) {
		track_url = track_url + ' (' + extension + ')';
	    }
	    track_url = track_url + '?' + this.segmentation_variables;
	    this.region_tracker._trackPageview(track_url);
	}
        if (this.seller_tracker) {
            var track_url = url;
            if (extension) {
		track_url = track_url + ' (' + extension + ')';
            }
	    track_url = track_url + '?' + this.segmentation_variables;
            this.seller_tracker._trackPageview(track_url);
        }

	for (i in this.group_trackers) {
	    var track_url = detail_url || detail_pagename || url;
	    track_url = track_url;

	    if (extension) {
		track_url = track_url + ' (' + extension + ')';
	    }
	    track_url = track_url + '?' + this.segmentation_variables;
	    this.group_trackers[i]._trackPageview(track_url);
	}
    };
    this.track_it = function(el) {
        // console.log('track_it: el = |' + describe_elem(el) + '|');
        if (el.nodeType == 3) el = el.parentNode;
        while (!el.className || !el.className.match(/\btrackable\b/))
            el = el.parentNode;
        var page_url = el.getAttribute('trackurl');
        // console.log('trackurl |' + page_url + '|');
        if (!page_url) {
            var track_ext = el.getAttribute('trackext');
            page_url = detail_pagename + (track_ext ? ' ' + track_ext : '');
        }
        // alert('track_it: tracking |' + page_url + '|');
        MultiplyTracker.track_url(page_url);
    };
    this.setup_trackables = function() {
        var trackables = getElementsByClassName('trackable');
        for (var i in trackables) {
            if (trackables[i]) {
                addEvent(trackables[i], "click", function(e) {
                    if (!e) var e = window.event;
                    var target = e.target ? e.target : e.srcElement;
                    if (target)
                        this.track_it(target);
                }, false);
                // console.log('setup_trackables: set up for ' + describe_elem(trackables[i]) + ' done.');
            }
        }
    };
    this.set_custom_var = function(id,label,value,scope) {
        this.main_tracker._setCustomVar(id,label,value,scope);
        if (this.seller_tracker) {
	    this.seller_tracker._setCustomVar(id,label,value,scope);
        }
        if (this.region_tracker) {
	    this.region_tracker._setCustomVar(id,label,value,scope);
        }
    };
}
