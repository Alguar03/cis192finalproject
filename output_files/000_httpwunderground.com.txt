		var grouptitle = $('.groupTitle',$(this));
		var fulllist = $('.fullList',$(this));
		var inactive = false;
		if(!$(this).hasClass('activeGroup')) inactive=true;
		grouptitle.mouseenter(function(){
		fulllist.removeClass('menuHide').addClass('menuShow');
		if(inactive) container.addClass('activeGroup');
		});
		$(this).mouseleave(function(){
		fulllist.addClass('menuHide').removeClass('menuShow');
		if(inactive) container.removeClass('activeGroup');
		});
		grouptitle.click(function(e) {
		if (fulllist.hasClass('menuShow')) {
		// link behavior is okay
		}
		else {
		// cancel the click event
		e.preventDefault();
		// show the menu
		fulllist.removeClass('menuHide').addClass('menuShow');
		if(inactive) container.addClass('activeGroup');
		}
		});
		});
	})(jQuery);
