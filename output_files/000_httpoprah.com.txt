// Must use plain-old-javascript (no Prototype)
function setTopHeader() {
  var _login = [], _greeting = [], _username = [];
  var _each = Array.prototype.forEach;

  function each(iterator) {
    for (var i = 0, length = this.length; i < length; i++)
      iterator(this[i]);
  }
  if (!_each) _each = each;
  
  if (typeof document.getElementsByClassName == 'function') {
    _login = document.getElementsByClassName('top_login');
    _greeting = document.getElementsByClassName('top_greeting');
    _username = document.getElementsByClassName('username_slot');
  }
  else {
    var allTags = document.getElementsByTagName('*'),
        len = allTags.length;

    while (len--) {
      var el = allTags[len];
      if (el.className.indexOf('top_login') >= 0) {
        _login.push(el);
      } else if (el.className.indexOf('top_greeting') >= 0) {
        _greeting.push(el);
      } else if (el.className.indexOf('username_slot') >= 0) {
        _username.push(el);
      }
    }
  }

  _login.each = _each;
  _greeting.each = _each;
  _username.each = _each;

  // get cookie helper
  function getC(name) {
    if (document.cookie) {
      var parts = document.cookie.match(new RegExp('(?:^|[\\s;])' + name + '=([^;]*)'));
      return parts ? decodeURIComponent(parts[1]) : null;
    }
  }

  subsLink = 'http://myown.oprah.com/profile/newsletter.html';
  if (getC('AUTH')) {
    var nameText = getC('DISPLAY_NAME'),
        entityId = getC('ENTITY_ID');

    _login.each(function(el) {
      el.style.display = 'none';
    });

    _greeting.each(function(el) {
      el.style.display = 'block';
    });

    _username.each(function(el) {
      var a = el.getElementsByTagName('a')[0];
      a.innerHTML = nameText;
      a.href = 'http://myown.oprah.com/community/library/activity.html?entity_id=' + entityId;
    });

    // Add dynamic meta tag for Omniture
    var meta = document.createElement('meta');
    meta.name = 'username';
    meta.content = nameText;
    document.getElementsByTagName('head')[0].appendChild(meta);
  }
  else {
    _login.each(function(el) {
      el.style.display = 'block';

      var a = el.getElementsByTagName('a');
      if (a[0].href.indexOf(location.host + location.pathname) >= 0) {
        el.style.display = 'none';
      }
      });

    _greeting.each(function(el) {
      el.style.display = 'none';
    });

    if (window.logIn && window.loginCheck) {
      subsLink = "javascript:void(loginCheck({login_req:1, onLogin:function(){location.href='" + subsLink + "'}}))";
    }
    else {
      //subsLink = 'http://myown.oprah.com/register.html?redir=' + encodeURIComponent(subsLink);
      subsLink = 'http://myown.oprah.com/register.html';
    }
  }
  var moreLink = document.getElementById('ocom_more_subscribe');
  if (moreLink) {
    moreLink.href = subsLink;
  }

  // Fire login event for whoever's listening
  // onFire(bkim+schun)
  if (window.Prototype && document.fire) {
    document.fire('maui:login');
  }
}

      docElem = document.documentElement;

    while (el && el != context) {
      rt.y += el.offsetTop;
      rt.x += el.offsetLeft;
      el = el.offsetParent;
    }

    return rt;
