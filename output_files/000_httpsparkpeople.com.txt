{
	var strValidChars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789_-';
	for(i = 0; i < strParam.length; i++)
	{
		if (strValidChars.indexOf(strParam.charAt(i)) < 0)
		{
			return false;
		}
	}
	return true;
}
