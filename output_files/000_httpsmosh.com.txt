function handleCarouselMouseOver(element) {
      var   slideshow = jQuery("#smosh_slideshow"),                     
            link = jQuery("#smosh_slideshow_data a.slideshow_link").clone(),
            large_image = link.find("img");           
      if (jQuery("#smosh_slideshow").find("a").attr("href") !== element.attr("link") ) {                                                              
            slideshow.find("img").fadeOut("slow");
            link.attr("href", element.attr("link") ); 
            large_image.attr("src", element.attr("large_image") );            
            slideshow.html("").append(link);
            link.fadeIn("slow");
            // the image is active, for now
            jQuery("#smosh_carousel img").removeClass("active");              
            element.addClass("active");               
      }                 
};

jQuery(document).ready(function() {

      jQuery(slidshow_data).each( function(index, item) {

            list_item.find("a")
                  .attr("href", item.link);
            list_item.find("img")
                  .attr("src", item.small_image )

                  .attr("link", item.link);
            list_item.appendTo(list);

      });   
      
      smosh_carousel.find("img").bind("carousel_mouseover",  function() {                             
            var element = jQuery(this);
            handleCarouselMouseOver(element);

                  slideShowActiveIndex = 1;
            }                             
            smosh_carousel.find("img:eq(" + (slideShowActiveIndex-1) + ")").trigger("carousel_mouseover");                                                       
      }, 5000);
