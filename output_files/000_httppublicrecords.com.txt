	function showSearchingBox(e) {
		var s = document.getElementById('searching');

		s.style.left = f.offsetLeft;
		s.style.top = f.offsetTop;
		s.style.width = f.offsetWidth;
		//s.style.height = f.offsetHeight;
		s.style.visibility = 'visible';
		s.style.zIndex = 100;
		f.style.visibility = 'hidden';
		var d =document.getElementById('ctl00_ContentArea__searchForm__divLocationSummary');
		if (d!=null)
			d.style.display = 'none';
		d = document.getElementById('left-column');
		if (d!=null)
			d.style.visibility = 'hidden';
	}

	function narrowLocationData() {
		var ddlID = 'ctl00_ContentArea__searchForm_ctl00__ddlStates';
		if (ddlID.toString().length > 0) {
			var ddl = document.getElementById(ddlID);
			var tbl = document.getElementById("location-summary-table");
			if (ddl != null && tbl != null) {
				ddl.onchange = narrowLocationData;
				var st = ddl.options[ddl.selectedIndex].text.toLowerCase();
				for (idxRow = 0; idxRow < tbl.rows.length; idxRow+=2) {
					var cellst =tbl.rows[idxRow].cells[0].innerHTML.toLowerCase();
					var disp = (ddl.value == "" || cellst == st) ? "":"none";
					tbl.rows[idxRow].style.display = tbl.rows[idxRow + 1].style.display = disp;
				}
			}
		}
	}

					ddl.selectedIndex = idx;
					break;
				}
			}
			document.getElementById('ctl00_ContentArea__searchForm__btnSearch').click();			
		}
	}
