function bySortedValue(obj, context) {
var tuples = [];
for (var key in obj) tuples.push([key, obj[key].profession_name, obj[key].image, obj[key].profession_text]);
tuples.sort(function(a, b) { return a[1] > b[1] ? 1 : a[1] < b[1] ? -1 : 0 });
return tuples;
}

function bySortedCategory(obj, context){
  var tuples = [];
  for (var key in obj) tuples.push([key, obj[key].profession_name, obj[key].image, obj[key].description]);
  tuples.sort(function(a, b) { return a[1] > b[1] ? 1 : a[1] < b[1] ? -1 : 0 });
  return tuples;
}

function getProfessionByCampaign(){
    $.ajax( {
        type : "GET",
        url : "index.php",
        data : "module=dbviewdata&hive_op=professionByCampaign&campaign_id=" + $('#campaign_id').val(),
        success : function(data) {
            if(data != ''){
                decoded_data = jQuery.parseJSON(data);

                if(decoded_data.parent_id == '806'){
                    $("#company_text").html(decoded_data.profession_name);
                    $('#company-logo-div').css({'background-image': 'url("' + decoded_data.image + '")'});
                    $('#company-logo-div').css({'height': '190px'});
                    $('#company-logo-div').css({'width': '190px'});
                    $("#company-logo-div").show();

                    $('.searchbox').css({"margin-left": "60px"});
                    $("#search_header").html(decoded_data.profession_name);
                    $("#description").html(decoded_data.profession_text);
                    $("#category-title").html("About " + decoded_data.profession_name);
                    states = true;
                } else {
                    if(decoded_data.parent_id == '2019'){
                        //internships
                        $("#job_header").html('Internships');

                    } else if(decoded_data.parent_id == '1968') {
                        $("#job_category_label").hide();
                        $("#job_category").hide();
                        $("#job_title_label").html('Industry:');
                        $("#job_header").html('Franchise Opportunities');
                        $("<option>").attr("value", '1968').text('Franchise Oppurtunities').appendTo("#job_category");
                    } else if(decoded_data.parent_id == '1458') {
                        $("<option>").attr("value", '1458').text('Work From Home').appendTo("#job_category");
                        $("#job_category_label").hide();
                        $("#job_category").hide();
                        $("#job_title_label").hide();
                        $("#job_title").hide();
                        $('#zip_label').css({'margin-left': '80px'});
                        $('#zip').css({'margin-left': '80px'});
                        $('#zip').css({'width': '120px'});
                        $('#searchbox-first').css({'width': '415px'});
                        $('#searchbox-first').css({'margin-left': '120px'});
                    } else if(decoded_data.parent_id == '789') {
                        $("#job_category_label").hide();
                        $("#job_category").hide();
                        $("#job_title_label").html('Company:');
                        $("<option>").attr("value", '789').text('Company').appendTo("#job_category");

                        $("#job_header").html('Jobs');
                    }
                    $("#job_category").val(decoded_data.parent_id);
                    if($("#job_category").val() != ''){
                        $("#category-title").html($('#job_category option:selected').text());
                        changeCategoryImage();
                    }
                    matched_program = loadJobTitles(decoded_data.profession_id);
                }
            }

            $('#searchbox-first').show();
        }

    });

}

function loadJobCategories(job_category,job_title) {

  var matched_category = '';
  $.ajax( {
            type : "GET",
            url : "index.php",
            data : "module=dbviewdata&hive_op=job_categories&output=1",
            success : function(data) {
                    decoded_data = jQuery.parseJSON(data);
                    decoded_data_sort = bySortedCategory(decoded_data, this);
                    $.each(decoded_data_sort, function(key, value) {
                        // console.debug(key + "=>" + value);
                        $("<option>").attr("value", value[0]).text(value[1]).appendTo("#job_category");
                        if(value[0] == job_category){
                            matched_category = value[1];
                        }
                        images_by_cat['"' + value[0] + '"'] = value[2];
                        descr_by_title['"' + value[0] + '"'] = value[3];
                    });

                    $("#job_category").val(job_category);
                    changeCategoryImage();
                    changeDescription();

                    if($("#job_category").val() != ''){
                        matched_program = loadJobTitles(job_title);
                    }

                    if($("#job_category").val() == '2019'){
                        //internships
                        $("#job_header").html('Internships');

                    } else if($("#job_category").val() == '1968') {
                        $("#job_header").html('Franchise Opportunities');
                    } else {
                        $("#job_header").html('Jobs');
                    }

                    if(typeof $('#pre_selectedJobCategory').val() != "undefined" && $('#pre_selectedJobCategory').val() != 0 && $('#hive_op').val() != 'pathComplete'){
                        $('#job_category').hide();
                        $('#job_category_label').hide();
                    }

                    if(pre_selectedJobCategory == '' && $('#campaign_id').val() != ''){
                        getProfessionByCampaign();
                    } else {
                        $('#searchbox-first').show();
                    }

            }
        });
}
function loadJobTitles(job_title) {
    $('#job_title').html('');
    var matched_program = '';
    var required = false;
    //check if this is a required field if not change text to choose one
    var classList =$('#job_title').attr('class').split(/\s+/);
    $.each( classList, function(index, item){
        if (item === 'required') {
            required = true;
         }
     });
    if(required){
        $("<option>").attr("value", "").text("Choose One").appendTo("#job_title");
    } else {
        $("<option>").attr("value", "").text("See All").appendTo("#job_title");
    }
        $.ajax( {
            type : "GET",
            url : "index.php",
            data : "module=dbviewdata&hive_op=job_titles&output=1&job_category=" + $("#job_category").val(),
            success : function(data) {
                    decoded_data = jQuery.parseJSON(data);
                    val = $("#job_category").val();
                    dereferenced_data =  bySortedValue(decoded_data[val], this);
                    $.each(dereferenced_data, function(key, value) {
                        $("<option>").attr("value", value[0]).text(value[1]).appendTo("#job_title");
                        if(value[0] == job_title){
                            $("#search_header").html(value[1]);
                        }
                        //build image allocation array
                        images_by_title['"' + value[0] + '"'] = value[2];
                        descr_by_title['"' + value[0] + '"'] = value[3];
                    });

                    $("#job_title").focus();
                    $("#job_title").val(job_title);
                    changeImage(true);
                    changeDescription();

            }
        });
}
function changeImage(move_searchbox){

            $('#company-logo-div').css({'background-image': 'url("' + images_by_title['"' + $("#job_title").val() + '"'] + '")'});
            $('#company-logo-div').css({'height': '190px'});
            $('#company-logo-div').css({'width': '190px'});
            $("#company-logo-div").show();
            if(move_searchbox){

                $('.searchbox').css({"margin-left": "60px"});
            }
        }

        //$('.searchbox').css({'margin-left': '20px'});
    }
}

}

Why stay in a career that drains you when you can enroll in a certificate or degree program that can- literally - change your life for the better!? ");
    }
}
