	   function createInstallationGuide() {
			var body = document.getElementsByTagName("body")[0];
			var div = document.createElement("div");
			div.setAttribute("class", "installation-guide-bg")
			body.appendChild(div);
			var div_container = document.createElement("div");
			div_container.setAttribute("class", "installation-guide-img-container")
		//	var img = document.createElement("img");

			var img = document.getElementById("installation-guide-img")
			img.setAttribute("class", "installation-guide-img")
			img.setAttribute("id", "installation-guide-img")
			img.addEventListener("mousedown",function(e) {
				body.removeChild(div);
				div_container.removeChild(e.currentTarget);
			},false);
		//	img.setAttribute("src","/images/installationGuide.png")
			div_container.appendChild(img);
			img.style.display ="block"
			body.appendChild(div_container);
			img.style.marginTop = (document.documentElement.scrollTop+10) +"px"
			var interval;
			var opacity = 0;
			var opacityDelta = 0.03
			var sizeInterval = function(div) {
				  opacity += opacityDelta;
				  div.style['opacity'] = opacity;
				  finalOpacity = 0.8;
				  if (div.id=="installation-guide-img") finalOpacity = 1
				  if (opacity>=finalOpacity) {
					  clearInterval(interval)
					  interval = null;	  
					  if (div.id!="installation-guide-img") {
					     opacity = 0;
						 interval = setInterval(sizeInterval, 1, img);
					  }
				  }
			}
			interval = setInterval(sizeInterval, 1, div);
	   }

FoxTab powers Firefox with the following features:<br>
&nbsp;&nbsp;– Top Sites (aka Speed Dial) for quickly accessing your favorite web sites<br>
&nbsp;&nbsp;– Tab Flipper – to easily flip between opened tabs using mouse or keyboard gestures<br>
&nbsp;&nbsp;– Recently Closed Tabs – for reopening a tab that was recently closed<br>
<br>
As the web is rapidly turning into an application environment where dozens of tabs are being opened
concurrently, accessing a page visually is in times the fastest and best possible option.<br>
FoxTab introduces to Firefox the best and most attractive way for accessing the web as never seen
before on any other browser or operating system.
