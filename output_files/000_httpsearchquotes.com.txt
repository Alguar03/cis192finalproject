			var newNotifications = new Array();
			function checkForNewMessages() {
				$.ajax({
					url: 'http://www.searchquotes.com/notifications/getnewnotifications.php?shown=0',
					cache: false,
					success: function(data) {
						var returnObj = JSON.parse(data);
						if (returnObj.count > 0) {
							alertsAr = new Array(); 
							$.each(returnObj.results, function(key, value) {
								alertsAr[alertsAr.length] = value;
								newNotifications[newNotifications.length] = value.notificationId;
							});
							startFloatingAlerts(alertsAr);
							updateNewMessagesCount();
						}
					},
					complete: function(a, b) {
						setTimeout(function() {
						      checkForNewMessages();
						}, 20000);	
					}
				});			
			}

			function checkForAlertedMessages() {
				$.ajax({
					url: 'http://www.searchquotes.com/notifications/getnewnotifications.php?shown=1',
					cache: false,
					success: function(data) {
						var returnObj = JSON.parse(data);
						if (returnObj.count > 0) {
							$.each(returnObj.results, function(key, value) {
								newNotifications[newNotifications.length] = value.notificationId;
							});
							updateNewMessagesCount();
						}
					}
				});				
			}

			function startFloatingAlerts(alerts) {
				var i = 0;
				$.each(alerts, function(key, value) {
					$('body').append('<a href="javascript:void(0);" onClick="javascript:deleteNotifications();" class="floatingAlert"><div id="floatingAlert'+value.notificationId+'" class="floatingAlert"><b>'+value['from']+'</b> sent you a message:<br />"'+value.messageSubject+'"</div></a>');
					var vOffset = -20 - (150 + 20) * i;
					i++;
					$(document).scroll();
					$('#floatingAlert'+value.notificationId).position({
						my: "right bottom",
						  at: "right bottom",
						  of: $(window),
						  offset: "-20 "+ vOffset
					});
					$(document).scroll(function() {
						$('#floatingAlert'+value.notificationId).position({
							my: "right bottom",
							  at: "right bottom",
							  of: $(window),
							  offset: "-20 "+ vOffset
						});
					});
					$('#floatingAlert'+value.notificationId).animate({
						opacity: 1
						}, 3000, function() {
						setTimeout(function() {
							$('#floatingAlert'+value.notificationId).animate({
								opacity: 0
								}, 2000, function() {
							$('#floatingAlert'+value.notificationId).remove();
							});						
						}, 5000);				
					});					
				});
			}

			function updateNewMessagesCount() {
				var n = newNotifications.length;
				if (n == 0) {
					$('#newMessagesCountText').text('');
					$('#newMessagesCount').css('background-image', "url('./message_off.png')");
				}
				else {
					$('#newMessagesCountText').text(n);
					$('#newMessagesCount').css('background-image', "url('./message_on.png')");
				}
				$('#newMessagesCountText').css('background-color', '#f02525');
			}

			function deleteNotifications() {
				var ids = '(';
				$.each(newNotifications, function(key, value) {
					ids += value+',';
				});
				ids += '0)';
				$.ajax({
					url: 'http://www.searchquotes.com/notifications/getnewnotifications.php?action=delete&ids='+ids,
					cache: false,
					success: function(data) {
						window.location="/inbox/";
					}
				});
			}

				checkForAlertedMessages();
				checkForNewMessages();
			});
