function modifyNodeForIE6(node)
{
	node.onmouseover=function()
	{
	    if (this.className.indexOf(" over") == -1)
		    this.className+=" over";
	}
	if (navigator.appName != 'Microsoft Internet Explorer')
	{
	    node.onmouseout=function() 
	    {
		    this.className=this.className.replace(" over", "");
	    }
	}
	else
	{
	    node.onmouseleave=function() 
	    {
		    this.className=this.className.replace(" over", "");
	    }
	}
}

function processMenu(navRoot)
{
   	for (i=0; i<navRoot.childNodes.length; i++)
	{
		node = navRoot.childNodes[i]; //process first level menu
		if (node.nodeName=="LI") 
		{
		    modifyNodeForIE6(node);
		    for (x=0; x<node.childNodes.length; x++)
		   {
		        node2 = node.childNodes[x];
		        if (node2.nodeName=="UL") //process the 2nd level menu to allow 3rd level hover popout
		        {
		            for (y=0; y<node2.childNodes.length; y++)
		            {
		                node3 = node2.childNodes[y];
		               if (node3.nodeName=="LI")
		                modifyNodeForIE6(node3); 
		            }
		        }
		    }  
		} 
	}
}
