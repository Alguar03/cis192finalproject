<script type="text/javascript">
/* TIME ZONE BUTTON ACTIONS */
NBC('.timezone-tabs li a').click(function(e){
        var $tabsNode, $tab = NBC(this).parent();
        var timezone = NBC(this).html().toLowerCase();

        $tabsNode = $tab.closest(".timezone-tabs");
        $tabsNode.find(".active").removeClass("active");
        $tab.addClass("active");

        arrDomain = domain.split('.');
        rootDomain = arrDomain.slice(-2).join('.');

        NBC.cookie("g-tz", timezone, { expires: 365, path: '/', domain: rootDomain });
        NBC(".tonight .lineup-thumbs").hide();
        initScheduleCarousel(timezone);
});

var userTimezone = (NBC.cookie("g-tz")) ? NBC.cookie("g-tz") : "et";
NBC(".timezone-tabs li." + userTimezone + " a").trigger('click');

/* SCHEDULE CAROUSEL */
function initScheduleCarousel(timezone) {
        NBC("#tonight-carousel-" + timezone).show();

		NBC(document).ready(function(){
			$lineUpNext = NBC('#lineup-next-' + timezone), $lineUpPrev = NBC('#lineup-prev-' + timezone), lineUpCount = NBC("#tonight-carousel-" + timezone + " ul li").length;
			NBC("#tonight-carousel-" + timezone).jcarousel({
					scroll : 5
					,size: lineUpCount
					,buttonNextHTML : null
					,buttonPrevHTML : null
					,initCallback : function(carousel) {
							$lineUpNext.click(function() {
									carousel.next();
									return false;
							});
							$lineUpPrev.click(function() {
									carousel.prev();
									return false;
							});
					}
					,itemFirstInCallback : function(carousel, item, idx, state){
							if(state != "init") {
									if(idx == 1) $lineUpPrev.addClass("btn-disabled");
									else $lineUpPrev.removeClass("btn-disabled");
							}
					}
					,itemLastInCallback : function(carousel, item, idx, state){
							if(idx == lineUpCount) $lineUpNext.addClass("btn-disabled");
							else $lineUpNext.removeClass("btn-disabled");
					}
			});
		});
}
