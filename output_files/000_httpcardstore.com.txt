		if(isMember){
			cookieData["member_status_wl"] = "1";
		}else if(isRegistered){
			cookieData["member_status_wl"] = "2";
		}else{
			cookieData["member_status_wl"] = "3";
		}
		//If user browsed cards
		if(csopts.pageType == 'gallery' || csopts.pageType == 'landing'){
			cookieData["browsed_paper"] = "1";
		}
	}
	var cookieValue = $.param(cookieData);
	$.cookie(cookieName, cookieValue, { expires: (365*26), domain: '.americangreetings.com', path: '/'});
}

function getCookieData(cookieVar){
	var cookieKeyVal = cookieVar.split("&");
	var cookieData = {};
	$.each(cookieKeyVal, function(x,y){
	    var temp = y.split('=');
	    cookieData[temp[0]] = temp[1];
	});
	return cookieData;	
}

function getEnvPrefix(){
	var location = document.URL;
	//http://cardstore.local
	location = location.substring(0,22);
	if(location.indexOf("local") > 0){
		return "dev";
	}else if(location.indexOf("stage") > 0){
		return "stage";
	}else if(location.indexOf("work") > 0){
		return "work";
	}else{
		return "www";
	}
}
