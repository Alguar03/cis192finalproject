	var imageRotator = Class.create({
	initialize: function(fadeDuration, displayDuration, anchorElement, foregroundElement, backgroundElement, images) {
		/* seconds the actual fade should last */
		this.fadeDuration      = fadeDuration;
		/* seconds the image is shown for before next fade begins */
		this.displayDuration   = displayDuration;
		/* should be an IMG element prepopulated with 1st image of rotation */
		this.foregroundElement = foregroundElement;
		/* should be an IMG element */
		this.backgroundElement = backgroundElement;
		/* should be a Anchor element */
		this.anchorElement = anchorElement;
		/* array of image URLs used for rotation */
		this.images            = images;
	},
	fader: function(currentIndex, effect) {
		var nextIndex   = currentIndex + 1;
		var nextEffect  = 'fadeOut';
		var divToChange = this.foregroundElement;

		if (effect == 'fadeOut') {
			nextEffect  = 'fadeIn';
			divToChange = this.backgroundElement;
		}

		if (nextIndex > this.images.length - 1) nextIndex = 0;


		var x = '';

		if (this.is_mac_chrome) {
			x = '?x=' + Math.floor(Math.random()*1000000);
		}


		divToChange.setAttribute('src', this.images[nextIndex].image_url + x);

		var iRObj = this;

		setTimeout( function() {
				var landingPage = iRObj.images[nextIndex].pic_url;
				if(landingPage) iRObj.anchorElement.setAttribute('href', landingPage);
				if (effect == 'fadeOut') {
					iRObj.foregroundElement.fadeOut( iRObj.fadeDuration,
						function() { iRObj.fader( nextIndex, nextEffect ) },
						'easeOutExpo');
				} else {
					iRObj.foregroundElement.fadeIn( iRObj.fadeDuration,
						function() { iRObj.fader( nextIndex, nextEffect ) },
						'easeOutExpo');
				}
			},
			this.displayDuration * 1000
		);
	},
	start: function () {
		var ua = navigator.userAgent;
		var os = navigator.platform;
		this.is_mac_chrome = (ua.indexOf('Chrome') != -1 && os.indexOf('Mac') != -1) ? 1 : 0;
		this.fader(0, 'fadeOut');
	}
});

Element.addMethods(['DIV','SPAN','IMG'], {
	// fade an element out over {duration} seconds, call a callback when it's done
	fadeOut: function (element, duration, callback, easingType) {
		if (typeof(duration) === 'undefined') {
			duration = 0.5;	
		}
		return element._fade(duration, callback, true, easingType);
	},
	
	// fade an element in over {duration} seconds, call a callback when it's done
	fadeIn: function (element, duration, callback, easingType) {
		if (typeof(duration) === 'undefined') {
			duration = 1;	
		}
		return element._fade(duration, callback, false, easingType);
	},
	
	// helper method for fading
	// available easing types ();
	_fade: function (element, duration, callback, isFadeOut, easingType) {
		var fade_counter = 0;
		//Geometric decay interval
		var fade_interval = duration / 45;
		if (isNaN(fade_interval) || fade_interval < 0.01) { fade_interval = 0.01; }
		
		// if we are fading it in then make sure the element is 'visible' (eg display is not none)
		if (!isFadeOut && !element.visible()) {
			element.setOpacity(0);
			element.show();
		}
		
		//Fade out before removing from the DOM
		new PeriodicalExecuter(
			function(pe) {
				
				var fadeDelta = isFadeOut ? - 1 : 1;
				var fadeCheck = isFadeOut ? element.getOpacity() > 0 : element.getOpacity() < 1;
				
				//There is still something to fade
				if (fadeCheck) {
					if(!easingType){
						//Geometrically decay the image
						element.setOpacity(element.getOpacity() + .001 * (++fade_counter) * fadeDelta);
					}else{
						//Geometrically decay the image
						var startValue = isFadeOut?1:0;
						var endValue = isFadeOut?0:1;
						var b = startValue;
						var c = endValue - startValue;
						var d = duration;
						var z = endValue;
						var t = fade_counter  / 45;

						/* define few easing calculations we could use */
						/* todo: create custom easing class */
						var opac;
						if(easingType == 'easeOutExpo')
							opac = (t==d) ? b+c : c * (-Math.pow(2, -10 * t/d) + 1) + b;
						else if(easingType=='easeInExpo')
							opac = (t==0) ? b : c * Math.pow(2, 10 * (t/d - 1)) + b;
						else if(easingType=='easeInOutExpo'){
							if (t==0) opac = b;
							if (t==d) opac = b+c;
							if ((t/=d/2) < 1) opac = c/2 * Math.pow(2, 10 * (t - 1)) + b;
							opac = c/2 * (-Math.pow(2, -10 * --t) + 2) + b;
						}
						else if(easingType=='easeInOutSine')
							opac = -c/2 * (Math.cos(Math.PI*t/d) - 1) + b;
						else if(easingType=='easeOutSine')
							opac = c * Math.sin(t/d * (Math.PI/2)) + b;
						else if(easingType=='easeInSine')
							opac = -c * Math.cos(t/d * (Math.PI/2)) + c + b;
						else
							opac = c*(t/=d)*t + b;
						element.setOpacity(opac);
						fade_counter++;
					}
				//Nothing is visible
				} else {
					element.setOpacity(!isFadeOut);
					//Stop the fading
					pe.stop();
					
					// show/hide the element
					element[isFadeOut ? 'hide' : 'show']();
					
					if (callback) {
						callback();
					}
				}
			},
			fade_interval
		);
		
		return element;
	}
});
