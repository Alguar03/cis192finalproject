Tax.FFATestHelper.ffa_tts_test.getAbTestParam 	= function() {
	if(Tax.FFATestHelper.ffa_tts_test.enabled) {
		var rand 	= Math.floor(Math.random()*9 + 1); // generate random number from 1 - 9
		var recipe 	= undefined;						
				
		switch(rand) {
			case 1:
				recipe = 'A';
				break;
			case 2:
				recipe = 'B';
				break;
			case 3:
				recipe = 'C';
				break;
			case 4:
				recipe = 'D';
				break;
			case 5:
				recipe = 'E';
				break;
			case 6:
				recipe = 'F';
				break;
			case 7:
				recipe = 'G';
				break;
			case 8:
				recipe = 'H';
				break;
			case 9:
				recipe = 'I';
				break;
			default:
				break;
		}

		var abtestQueryString	= '';
		
		if(recipe) {
			var abtestParam	= 'tts';
			abtestQueryString = abtestParam + "=" + recipe;					
			Tax.FFATestHelper.ffa_tts_test.eVar43 = "ty11_2450_tts_" + recipe;			
		}

		return abtestQueryString;
		
	} else {
		return '';
	}
};
