function getRules(){
	if (!document.styleSheets) return;
	
	if (document.styleSheets[1].cssRules)
		return document.styleSheets[1].cssRules;
	else if (document.styleSheets[1].rules)
		return document.styleSheets[1].rules;
	else return;
}	

function replaceBackgroundImage(newurl){
	theRules = getRules();
	theRules[7].style.background = "url(/img/home/main/"+newurl+") no-repeat scroll 0px -8px";
	
	theRules[8].style.background = "url(/img/home/main/"+newurl+") no-repeat scroll center 22px";
}

function finalize(){

function parseHelper(str, t){
	s = str.substr(str.indexOf(t) + t.length);
	s = s.substr(0, s.indexOf('|'));
	return s;
}

function parseData(str){
	replaceSpan1(parseHelper(str, "title_v="));
	replaceSpan2(parseHelper(str, "title_h="));
	replaceSpan3(parseHelper(str, "title_width="));
	replaceP1(parseHelper(str, "caption_v="));
	replaceP2(parseHelper(str, "caption_width="));
	replaceP3(parseHelper(str, "caption_h="));
	replaceH3a(parseHelper(str, "title_h="));
	replaceH3b(parseHelper(str, "title_h="));
	replaceBackgroundImage(parseHelper(str, "main_img="));
}

	x_getCampaignSpace(cid, function(t){updateObject("sectiontitle", t)});
	x_getCampaignCaption(cid, "true", function(t){updateObject("campaignsection", t)});

		x_addNettrackerClickAjax(function(){return});
	if (ENABLE_CLICK_TRACKER){
		x_clickTracker("homepageajax", function(){return});
	}
}

	return false;
}
