	registerNS("ww.popup");
	ww.popup.catselect = new function(){
		
		var timeout = null;
		this.togglePopup = function(){
			$j('#ww-categoryPopup').css("left", ($j("#ww-categoryTextbox").offset().left - 5)).css("top", $j("#ww-categoryTextbox").offset().top);
			this.clearTimeout();
			$j('#ww-categoryPopup').toggle('fast');
		}
		this.select = function(code){
			$j('#ww-categoryTextbox').data('value',code).html($j('#ww-categoryPopupItem' + code).data('value'));
			$j('#ww-categoryPopup').hide('fast');
		}
		this.value = function(){
			return $j('#ww-categoryTextbox').data('value');
		}
		this.hidePopup = function(){
			timeout = window.setTimeout(function(){
				$j('#ww-categoryPopup').hide('fast');
			}, 200);
		}
		this.clearTimeout = function(){
			if(timeout != null){
				window.clearTimeout(timeout);
				timeout = null;
			}			
		}
	}
