cx.exp.widgetlibrary.search.common.container.subscribeAllEvents = function() {
    var namespace = cx.exp.widgetlibrary.search.common.container, radioButton; 
    var content = new Object();
	var defaultLob = 'flightOnly';
    //initial the selectedRadioButton (will be run once)
    cx.exp.widgetlibrary.search.common.container.selectedRadioButton = defaultLob;

    if (!(YAHOO.lang.isNull(namespace.titleDiv) || YAHOO.lang.isUndefined(namespace.titleDiv))) {
        content.title = namespace.titleDiv.innerHTML
    }
    if (!(YAHOO.lang.isNull(namespace.wizardDiv) || YAHOO.lang.isUndefined(namespace.wizardDiv))) {
    	content.wizard = namespace.wizardDiv.innerHTML
    }
    namespace.cacheWizardContent[defaultLob] = content;

	// Since this function is the entry point to wizard JS functionality,
	// let's whether we have a saved wizard state available (back button support)
	var history = YAHOO.util.Dom.get('uw_wizardstate').value;
	// preloadNonDefault will contain the wizard type that should be initially on the page IF it is other than the default wizard
	// otherwise (if we don't need to change the default shown wizard) it can be null.
	var preloadNonDefault = null;
	if (history){
		namespace.historyObject = new cx.exp.search.common.model(YAHOO.lang.JSON.parse(history));
		if (defaultLob == namespace.historyObject.lob){
			// Default wizard needs to be shown - already rendered on the page
			// Let's start up the wizard by calling it's initializer function (note the naming convention wizardType+'Fn')
			eval('namespace.wizardObj = '+namespace.historyObject.lob + 'Fn(namespace.historyObject);');
			
			 // for dropdown, initial the destination based on the current origin/initial origin to empty(default hint)
			cx.exp.widgetlibrary.search.common.dropdown.updateDestinationByOrigin();
		}
		else{
			// We need to show a wizard other than the default)
		    preloadNonDefault = namespace.historyObject.lob;
		}
	}
	else
	{
		// Previously saved wizard state not available. Default wizard will be shown.
		// Let's start up the wizard by calling it's initializer function (note the naming convention wizardType+'Fn')
		eval('namespace.wizardObj = '+'flightOnlyFn(null);');
	}

    var radioButtons = cx.exp.widgetlibrary.search.common.container.radioButtons;
    for (var i=0; i<radioButtons.length; i++) {
        var radioButton = radioButtons[i];
        if (!(YAHOO.lang.isNull(radioButton) || YAHOO.lang.isUndefined(radioButton))) {
               YAHOO.util.Event.addListener(radioButton, "click", namespace.changeWizardContent);
				if (preloadNonDefault === radioButton.getAttribute('wizardType')){
					// If we need to activate a wizard other than the default one, let's manually call the LOB change handler
					namespace.changeWizardContent.call(radioButton);
				}
				else if (!preloadNonDefault && radioButton.checked && defaultLob != radioButton.getAttribute('wizardType')){
					// If a radio button is pre-selected, make sure we have the right wizard
					namespace.changeWizardContent.call(radioButton);
				}
        }
    }
};
