  	function loadLibrary(lib, targetElement, callback) {
  		var script= document.createElement('script');
  		script.type= 'text/javascript';
  		script.src= lib;

  					callback();
  				}
  			}
  			script.onload = callback;
  		}
  		targetElement.appendChild(script);
  	}
