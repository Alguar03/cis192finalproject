        var checkedCreds = false;
        $('#logOn').submit(function () {
            if (checkedCreds) // bypass if we've already checked

                $('#validation-message').html("Please enter your username/email and password.");

            }
            else {
                // ajax authenticate before submitting
                $.ajax({
                    url: 'https://www.beyonddiet.com/BD/Account/Authenticate',
                    data: { userID: $('#UserName').val(), password: $('#Password').val() },
                    type: 'GET',
                    dataType: 'jsonp',
                    cache: false,
                    success: function (data) {
                        if (data.isValid) {
                            checkedCreds = true;
                            $('#logOn').submit();
                        }
                        else {
                            $('#validation-message').html(data.message);
                            showValidationMsg();
                        }
                    },
                    error: function (res) { // error event does not get triggered for jsonp requests
                        // log error
                        $.ajax({
                            url: '/Members/Home/LogError',
                            data: { message: res.responseText },
                            type: 'POST',
                            cache: false
                        });

                        // try to post anyway
                        checkedCreds = true;
                        $('#logOn').submit();
                    }
                });
            }

            return false;
        });
    });

            offsetTop: 10
        }).focus();
    }
